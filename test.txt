            function api_set_wordLength(str, length) {
                var substr;
                var word_sum = 0;
                var len = 0;
                //判断中文字符所占的空间大小
                for (var i = 0;i < str.length;i++) {
                    //中文字符
                    if (str.charCodeAt(i) > 127 || str.charCodeAt(i) == 94) {
                        len += 2;
                        word_sum++;
                    } else {
                        //英文字符
                        len++;
                        word_sum++;
                    }
                    if (len >= length && i != str.length - 1) {
                        return str.substr(0, word_sum) + '...';
                    }
                }
                return str;
            }
